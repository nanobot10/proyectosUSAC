#crear nuevo usuario

POST - http://localhost:5000/ayd/practica3/v1/auth/signup

request: 
{
	"name": "Test user",
	"username": "test123",
	"email": "test.user@gmail.com",
	"password": "password"
}
response: 
{
    "success": true,
    "message": "User registered successfully",
    "data": {
        "userCode": 1009
    }
}

# Hacer transferencia

POST - http://localhost:5000/ayd/practica3/v1/users/transfer
Header : Authorization Bearer login-token


request: 
{
	"accountNumber": "eea3bfa7-6d3a-477d-a148-e16a19e170d1",
	"amount": 150.0
}

response:
{
    "success": true,
    "message": "success",
    "data": {
        "newBalance": 850.0
    }
}

# Aprovar  o denegar un credito , Solo el usuario admin con rol ROLE_ADMIN puede acceder a este recurso, el estado puede ser APPROVED o DENIED

Header : Authorization Bearer login-token

        "id": 8,

        "status": "APPROVED"
    }
}
